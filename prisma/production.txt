List all tables:
----------------
SELECT table_name
FROM information_schema.tables
WHERE table_schema = 'public'
ORDER BY table_name;

Drop all tables:
----------------
select 'drop table if exists "' || tablename || '" cascade;' 
  from pg_tables
  where schemaname = 'public';

------------------------------------
